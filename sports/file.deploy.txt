#!/usr/bin/env bash
set -euo pipefail

# Sports package deploy script for deploy_helper.py.
# The helper replaces:
# {{PACKAGE_PATH}}
# {{EXTRACT_DIR}}
# {{SOURCE_SITE_DIR}}
# {{DEST_PARENT_DIR}}
# {{DEST_SITE_DIR}}
# {{SITE_NAME}}
# {{COMMIT_MESSAGE}}

echo "== Deploying {{SITE_NAME}} from {{PACKAGE_PATH}} =="
echo "Source: {{SOURCE_SITE_DIR}}"
echo "Dest:   {{DEST_SITE_DIR}}"

# Safe copy: do not use rsync --delete here.
# Keep Terraform state, local .tfvars, generated zip artifacts, and package-staged workflows out of the site copy.
rsync -a   --exclude '.terraform/'   --exclude 'terraform.tfstate'   --exclude 'terraform.tfstate.backup'   --exclude '*.tfstate'   --exclude '*.tfstate.backup'   --exclude 'api.zip'   --exclude 'ingest.zip'   --exclude '.github/'   "{{SOURCE_SITE_DIR}}/" "{{DEST_SITE_DIR}}/"

# GitHub only discovers workflows at the repository root: .github/workflows/.
# The package stages workflows at {{EXTRACT_DIR}}/.github so they land beside the sports/ folder in the repo.
if [ -d "{{EXTRACT_DIR}}/.github" ]; then
  echo "== Copying GitHub Actions workflows to repo root =="
  mkdir -p "{{DEST_PARENT_DIR}}/.github"
  rsync -a "{{EXTRACT_DIR}}/.github/" "{{DEST_PARENT_DIR}}/.github/"
elif [ -d "{{SOURCE_SITE_DIR}}/.github" ]; then
  echo "== Copying legacy packaged GitHub Actions workflows to repo root =="
  mkdir -p "{{DEST_PARENT_DIR}}/.github"
  rsync -a "{{SOURCE_SITE_DIR}}/.github/" "{{DEST_PARENT_DIR}}/.github/"
else
  echo "No packaged GitHub Actions workflows found."
fi

cd "{{DEST_PARENT_DIR}}"

echo "== Validation =="
python3 -m py_compile   "{{SITE_NAME}}/lambda/api/handler.py"   "{{SITE_NAME}}/lambda/ingest/handler.py"   "{{SITE_NAME}}/scripts/seed_data.py"

if command -v node >/dev/null 2>&1; then
  node --check "{{SITE_NAME}}/site/app.js"
else
  echo "node not found; skipping frontend syntax check."
fi

if command -v terraform >/dev/null 2>&1; then
  terraform -chdir="{{SITE_NAME}}/terraform" fmt -recursive
  terraform -chdir="{{SITE_NAME}}/terraform" init -backend=false
  terraform -chdir="{{SITE_NAME}}/terraform" validate
else
  echo "terraform not found; skipping terraform fmt/init/validate."
fi

echo "== Git =="
git status
git add   "{{SITE_NAME}}/VERSION"   "{{SITE_NAME}}/README.md"   "{{SITE_NAME}}/Makefile"   "{{SITE_NAME}}/lambda"   "{{SITE_NAME}}/site"   "{{SITE_NAME}}/terraform"   "{{SITE_NAME}}/data"   "{{SITE_NAME}}/docs"   "{{SITE_NAME}}/scripts"   "{{SITE_NAME}}/deployer"   "{{SITE_NAME}}/file.deploy.txt"

if [ -d ".github/workflows" ]; then
  git add ".github/workflows"
fi

git status
git commit -m "{{COMMIT_MESSAGE}}"
git push
