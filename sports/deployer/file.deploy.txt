#!/usr/bin/env bash
set -euo pipefail

# Sports package deploy script for deploy_helper.py.
# The helper replaces:
# {{PACKAGE_PATH}}
# {{EXTRACT_DIR}}
# {{SOURCE_SITE_DIR}}
# {{DEST_PARENT_DIR}}
# {{DEST_SITE_DIR}}
# {{SITE_NAME}}
# {{COMMIT_MESSAGE}}

echo "== Deploying {{SITE_NAME}} from {{PACKAGE_PATH}} =="
echo "Source: {{SOURCE_SITE_DIR}}"
echo "Dest:   {{DEST_SITE_DIR}}"

# Safe copy: do not use rsync --delete here.
# Keep Terraform state, local .tfvars, and generated zip artifacts out of the copy.
rsync -a \
  --exclude '.terraform/' \
  --exclude 'terraform.tfstate' \
  --exclude 'terraform.tfstate.backup' \
  --exclude '*.tfstate' \
  --exclude '*.tfstate.backup' \
  --exclude 'api.zip' \
  --exclude 'ingest.zip' \
  --exclude '.github/' \
  "{{SOURCE_SITE_DIR}}/" "{{DEST_SITE_DIR}}/"

# If the package includes a workflow under sports/.github, copy it to the repo root.
# This keeps GitHub Actions in the normal .github/workflows location without deleting existing workflows.
if [ -d "{{SOURCE_SITE_DIR}}/.github" ]; then
  mkdir -p "{{DEST_PARENT_DIR}}/.github"
  rsync -a "{{SOURCE_SITE_DIR}}/.github/" "{{DEST_PARENT_DIR}}/.github/"
fi

cd "{{DEST_PARENT_DIR}}"

echo "== Validation =="
python3 -m py_compile \
  "{{SITE_NAME}}/lambda/api/handler.py" \
  "{{SITE_NAME}}/lambda/ingest/handler.py" \
  "{{SITE_NAME}}/scripts/seed_data.py"

if command -v node >/dev/null 2>&1; then
  node --check "{{SITE_NAME}}/site/app.js"
else
  echo "node not found; skipping frontend syntax check."
fi

if command -v terraform >/dev/null 2>&1; then
  terraform -chdir="{{SITE_NAME}}/terraform" fmt -recursive
  terraform -chdir="{{SITE_NAME}}/terraform" init -backend=false
  terraform -chdir="{{SITE_NAME}}/terraform" validate
else
  echo "terraform not found; skipping terraform fmt/init/validate."
fi

echo "== Git =="
git status
git add \
  "{{SITE_NAME}}/VERSION" \
  "{{SITE_NAME}}/README.md" \
  "{{SITE_NAME}}/Makefile" \
  "{{SITE_NAME}}/lambda" \
  "{{SITE_NAME}}/site" \
  "{{SITE_NAME}}/terraform" \
  "{{SITE_NAME}}/data" \
  "{{SITE_NAME}}/docs" \
  "{{SITE_NAME}}/scripts" \
  "{{SITE_NAME}}/deployer" \
  "{{SITE_NAME}}/file.deploy.txt"

if [ -f ".github/workflows/terraform-plan.yml" ]; then
  git add ".github/workflows/terraform-plan.yml"
fi

git status
git commit -m "{{COMMIT_MESSAGE}}"
git push
